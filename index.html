                    let num=1;
                    for(let i=0;i< N;i++){
                      num = num * array[i];
                    }
                    return num;
                  };

Q9. (Unique Color Shirt)

Ans. Input :- 6
                3 2 4 1 2 3     ||      Output:-  2

                function Unique_Shirts (arr,N) {
                    let sum = 0;
                      for(i = 0; i < N; i++){
                        for(j = 0; j < N; j++){
                          if(arr[i] == arr[j] && i != j){
                            break;
                            }
                          }
                      if(j ==N ){
                        sum++;
                      }
                    }
                    return sum;
                  }

Q10. (Min and Max) Congratulations on making up to this question. Let us give you a fairly simple array problem to solve. If you know how to iterate through the array, you will easily be able to solve this.

Ans. Input :- 6 3 1 4 6 2 7    ||      Output:-  17

                function arrayMin(arr) {
                    let min=arr[0]
                    for(let i=0;i< arr.length;i++){
                      if(arr[i]< min){
                        min=arr[i];
                      }
                    }
                    return min;
                  }
                  
                  function arrayMax(arr) {
                    let max=0
                    for(let i=0;i< arr.length;i++){
                      if(arr[i]>max){
                        max=arr[i]
                      }
                    }
                    return max;
                  }

Q11. (Birthday Game)

Ans. Input :- 5
                2 2 1 3 2
                4 2     ||      Output:-  2

                function Birthday_Game(arr,D ,M) {
                    let sum=0;
                    let value=0;
                   for(let i=0;i< arr.length-M+1;i++){
                     sum=0;
                     for(j=i;j< M+i;j++){
                       sum=sum+arr[j]
                     }
                     if(sum==D){
                       value++;
                     }
                   }
                   return value;
                  }
